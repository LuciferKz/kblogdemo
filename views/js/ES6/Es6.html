/*
    解构赋值必须结构一一对应

    数组的解构赋值
 */

var [d,e,f] = [1 ,2 ,3];

console.log(d,e,f); //1,2,3

//允许制定默认值

var [value = true] = [];

console.log(value) // true

//对象的解构赋值

var {g,h} = {g : 4,h : 5};

console.log(g,h) //4,5

var { i : j } = { i : 6,j : 7 }

// console.log(i); // i is not defined;

console.log(j); // 6;

var k;
// 已声明变量的解构赋值

// { k } = { k : 8 }; //错误写法，会报错

({ k } = { k : 8 }); //正确写法
